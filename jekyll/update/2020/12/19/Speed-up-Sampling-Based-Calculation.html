def eval_shap_parallel(x_train, y_train, x_test, y_test):
    thread_ids = [i for i in range(threads_num)]

    pool = Pool()
    func = partial(sub_task, x_train, y_train, x_test, y_test)
    ret = pool.map(func, thread_ids)
    pool.close()

    ret_arr = np.asarray(ret)
    return np.sum(ret_arr, axis=0)
